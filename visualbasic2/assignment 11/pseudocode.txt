 
        DECLARE str1, firstpart, secondpart : String
        DECLARE badpos, notpos : Integer

        str1 ← ""
        firstpart ← ""
        secondpart ← ""
        badpos ← 0
        notpos ← 0

        INPUT("Enter String: "),str1 
        
        badpos ← InStr(str1, "bad")
        notpos ← InStr(str1, "not")

        If notpos < badpos 
        Then
            firstpart = Left(str1, notpos - 1)
            secondpart = Right(str1, Len(str1) - (badpos + 2))

        End If
        OUTPUT(firstpart & "good" , secondpart)
